154double Graph::EdmondsKarp(Vertex* s, Vertex* t) {
155 double maxFlow = 0.0;
156 auto bfs = [this, &s, &t]() -> double {
157 for (auto v: vertexSet) {
158 v->setVisited(false);
159 }
160 std::queue<Vertex *> q;
161 s->setVisited(true);
162 q.push(s);
163
164 std::unordered_map<Vertex *, Edge *> prev;
165 while (!q.empty()) {
166 Vertex *currVertex = q.front();
167 q.pop();
168 if (currVertex == t) break;
169 for (auto adj: currVertex->getAdj()) {
170 if (adj->getDest()->isVisited() || adj->getWeight() - adj->getFlow() <= 0.0) {
171 continue;
172 }
173 adj->getDest()->setVisited(true);
174 prev[adj->getDest()] = adj;
175 q.push(adj->getDest());
176 }
177 }
178 if (prev.find(t) == prev.end()) return 0.0;
179 double bottleNeck = std::numeric_limits<double>::max();
180 for (auto e = prev[t]; e != nullptr; e = prev[e->getOrig()]) {
181 bottleNeck = std::min(bottleNeck, e->getWeight() - e->getFlow());
182 }
183 for (auto e = prev[t]; e != nullptr; e = prev[e->getOrig()]) {
184 e->setFlow(e->getFlow() + bottleNeck);
185 e->getReverse()->setFlow(e->getReverse()->getFlow() - bottleNeck);
186 }
187 return bottleNeck;
188 };
189 double flow;
190 while ((flow = bfs()) > 0.0) {
191 maxFlow += flow;
192 }
193 return maxFlow;
194}
