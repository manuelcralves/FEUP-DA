262vector<string> parse_csv_line(const std::string& line) {
263 vector<std::string> fields;
264 string field;
265 bool in_quotes = false;
266
267 for (size_t i = 0; i < line.length(); ++i) {
268 if (line[i] == '"') {
269 in_quotes = !in_quotes;
270 continue;
271 }
272 if (line[i] == ',' && !in_quotes) {
273 fields.push_back(field);
274 field.clear();
275 } else {
276 field += line[i];
277 }
278 }
279 fields.push_back(field);
280 return fields;
281}

516void minCostTrains(Graph& railway) {
517 // Get source station
518 std::string sourceName;
519 std::cout << "Enter source station name: ";
520 std::cin.ignore();
521 std::getline(std::cin, sourceName);
522 Vertex* source = railway.findVertex(sourceName);
523 if (source == nullptr) {
524 std::cout << "Source station not found." << std::endl;
525 return;
526 }
527
528 // Get destination station
529 std::string destName;
530 std::cout << "Enter destination station name: ";
531 std::getline(std::cin, destName);
532 Vertex* destination = railway.findVertex(destName);
533 if (destination == nullptr) {
534 std::cout << "Destination station not found." << std::endl;
535 return;
536 }
537
538 int maxFlow = 0;
539 double totalCost = 0.0; // Total cost of trains allocated along augmenting path
540
541 // Initialize residual graph with the same capacities as original graph
542 for (Vertex* v : railway.getVertexSet()) {
543 for (Edge* e : v->getAdj()) {
544 e->setFlow(0);
545 }
546 }
547
548 // Loop until there is an augmenting path from source to destination
549 while (bfs(source, destination, railway)) {
550 // Find the bottleneck capacity along the augmenting path
551 double bottleneck = INF;
552 Vertex* v = destination;
553 std::string optimalServiceType = ""; // Stores the optimal service type for the bottleneck capacity
554 while (v != source) {
555 Edge* e = v->getPath();
556 double remainingCapacity = e->getWeight() - e->getFlow();
557 std::string service = e->getService();
558 double cost = 0.0;
559 if (service == "STANDARD") {
560 cost = 2.0; // Standard service cost is 2 per train
561 } else if (service == "ALFA PENDULAR") {
562 cost = 4.0; // Alpha service cost is 4 per train
563 } else {
564 std::cout << "Invalid service type encountered: " << service << std::endl;
565 return;
566 }
567 // Calculate the cost of trains for the remaining capacity of the edge
568 double trainCost = cost * remainingCapacity;
569 // Update the optimal service type if the current service type has lower cost per train
570 if (optimalServiceType.empty() || trainCost < (cost * bottleneck)) {
571 bottleneck = remainingCapacity;
572 optimalServiceType = service;
573 }
574 v = e->getOrig();
575 }
576
577 // Update the flow along the augmenting path with the optimal service type
578 v = destination;
579 while (v != source) {
580 Edge* e = v->getPath();
581 e->setFlow(e->getFlow() + bottleneck);
582 e->getReverse()->setFlow(e->getReverse()->getFlow() - bottleneck);
583 v = e->getOrig();
584 }
585
586 maxFlow += bottleneck;
587 // Update the total cost with the cost of trains allocated using the optimal service type
588 if (optimalServiceType == "STANDARD") {
589 totalCost += 2.0 * bottleneck;
590 } else if (optimalServiceType == "ALFA PENDULAR") {
591 totalCost += 4.0 * bottleneck;
592 }
593 }
594
595 std::cout << "Max Flow between " << sourceName << " and " << destName << " is " << maxFlow << ", with Min Cost " << totalCost << "" << endl;
596}

598// Function to perform breadth-first search (BFS) to find an augmenting path
599bool bfs(Vertex* source, Vertex* destination, Graph& railway) {
600 for (Vertex* v : railway.getVertexSet()) {
601 v->setVisited(false);
602 v->setPath(nullptr);
603 }
604
605 std::queue<Vertex*> queue;
606 queue.push(source);
607 source->setVisited(true);
608
609 while (!queue.empty()) {
610 Vertex* v = queue.front();
611 queue.pop();
612
613 for (Edge* e : v->getAdj()) {
614 Vertex* u = e->getDest();
615 if (!u->isVisited() && e->getWeight() > e->getFlow()) {
616 u->setVisited(true);
617 u->setPath(e);
618 queue.push(u);
619 }
620 }
621 }
622
623 return destination->isVisited();
624}
